$rg = "test-seleniumdocker-rg"
$loc = "westeurope"
$plan = "test-seleniumdocker-plan"
$stor = "testseleniumdockerstor"
$fun = "test-seleniumdocker-func"
$docker_image = "blogdevaisecam3914543553.azurecr.io/selenium:latest"

az functionapp create --resource-group $rg --os-type Linux --plan $plan --deployment-container-image-name $docker_image --name $fun --storage-account $stor
